FORM MAKE_DATA .
          LOOP AT GT_SFLI INTO GS_SFLI.
               IF GS_SFLI-FLDATE+4(2) = SY-DATUM+4(2).
                   GS_SFLI-COLOR = 'C' && COL_TOTAL && '10'.
              ENDIF.
               IF GS_SFLI-PLANETYPE = '747-400'.
                   GS_COLOR-FNAME = 'PLANETYPE'.
                   GS_COLOR-COLOR-COL = COL_POSITIVE.
                   GS_COLOR-COLOR-INT = '1'.
                   GS_COLOR-COLOR-INV = '0'.
                    APPEND GS_COLOR TO GS_SFLI-IT_COLFIELDS.
              ENDIF.
              MODIFY GT_SFLI FROM GS_SFLI
              TRANSPORTING COLOR LIGHT IT_COLFIELDS.

              CLEAR GS_COLOR.
         ENDLOOP.
ENDFORM.
