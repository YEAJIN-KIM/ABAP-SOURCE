FUNCTION Z_BC400_SL_05_GOCONNECTION.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" VALUE(IV_CARRID) TYPE ZBC400_S_CONNECTION-CARRID
*" VALUE(IV_CONNID) TYPE ZBC400_S_CONNECTION-CONNID OPTIONAL
*" EXPORTING
*" REFERENCE(ES_CONNECTION) TYPE ZBC400_S_CONNECTION
*" EXCEPTIONS
*" NO_DATA
*"----------------------------------------------------------------------

* Global data declarations

SELECT SINGLE CARRID CONNID CITYFROM AIRPFROM
                            CITYTO AIRPTO FLTIME DEPTIME ARRTIME
              FROM SPFLI
              INTO ES_CONNECTION
              WHERE CARRID = IV_CARRID
              AND CONNID = IV_CONNID.

IF SY-SUBRC <> 0.
     RAISE NO_DATA.
ENDIF.

ENDFUNCTION.
