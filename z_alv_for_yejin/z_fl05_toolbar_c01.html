CLASS lcl_event_RECEIVER IMPLEMENTATION.

    METHOD ON_BEFORE_USER_COMMAND. "#### ### ##"
     PERFORM ON_BEFORE_USER_COMMAND USING E_UCOMM.

   ENDMETHOD.

 method on_toolbar. " ### ### ##"

   data ls_button type stb_button.
   ls_button-butn_type = '3'.
   INSERT ls_button INTO TABLE e_object->mt_toolbar.

   ls_button-function = 'PERCENTAGE'.
   ls_button-butn_type = '0'.
   LS_BUTTON-text = '% TOTAL'.
   LS_BUTTON-quickinfo = 'Total economy utilization'.
   INSERT ls_button INTO TABLE e_object->mt_toolbar.

   ls_button-function = 'PERCENTAGE_MARKED'.
   ls_button-butn_type = '0'.
   LS_BUTTON-text = '% marked'.
   LS_BUTTON-quickinfo = 'Economy utilization'.
   INSERT ls_button INTO TABLE e_object->mt_toolbar.
   ENDMETHOD.




  METHOD on_double_click. " # ## ### ##"
    DATA : LV_BOOKINGS_TOTAL TYPE I,
           LV_BOOKINGS_TOTAL_C TYPE C LENGTH 10,
           LV_MESSAGE_TEXT TYPE C LENGTH 60.

    READ TABLE GT_SFLIGHT INTO GS_SFLIGHT
    INDEX ES_ROW_NO-ROW_ID.

    LV_BOOKINGS_TOTAL = GS_SFLIGHT-SEATSOCC
    + GS_SFLIGHT-SEATSOCC_B + GS_SFLIGHT-SEATSOCC_F.

    LV_BOOKINGS_TOTAL_C = LV_BOOKINGS_TOTAL.
    LV_MESSAGE_TEXT = 'TOTAL NUMBER OF BOOKINGS : ' &&
    LV_BOOKINGS_TOTAL_C.

    MESSAGE LV_MESSAGE_TEXT TYPE 'I'.
  ENDMETHOD.

  METHOD on_user_command. " # ## ### ##"
    DATA LV_PERCENTAGE TYPE P LENGTH 3 DECIMALS 2. "### 3## ## 2##"
    DATA LV_CAPACITY TYPE I. "SEATMAX"
    DATA LV_OCCUPIED TYPE I. "SEATOCC"
    DATA LV_TEXT TYPE STRING.

    DATA : LT_ROW_NO TYPE LVC_T_ROID,
           LS_ROW_NO TYPE LVC_S_ROID,
           LS_COL TYPE LVC_S_COL.

    CASE E_UCOMM.

      WHEN 'DISP'.
      CALL METHOD go_alv->get_current_cell
        IMPORTING
          es_col_id = LS_COL
          es_row_no = LS_ROW_NO.
       LV_TEXT = '## ### #:' && LS_ROW_NO-ROW_ID.

       CONCATENATE LV_TEXT LS_COL-FIELDNAME "CONCATENATE : ### ### ###"
       INTO LV_TEXT SEPARATED BY SPACE. "SEPERATED BY : ### ### #####"

       MESSAGE LV_TEXT TYPE 'I'.

      WHEN 'PERCENTAGE'.

      LOOP AT GT_SFLIGHT INTO GS_SFLIGHT.
      LV_OCCUPIED = LV_OCCUPIED + GS_SFLIGHT-SEATSOCC
      + GS_SFLIGHT-SEATSOCC_B + GS_SFLIGHT-SEATSOCC_F.

      LV_CAPACITY = LV_CAPACITY + GS_SFLIGHT-SEATSMAX
      + GS_SFLIGHT-SEATSMAX_B + GS_SFLIGHT-SEATSMAX_F.
      ENDLOOP.

      LV_PERCENTAGE = LV_OCCUPIED / LV_CAPACITY * 100.
      MOVE LV_PERCENTAGE TO LV_TEXT.
      LV_TEXT = '## #### ###  ' && LV_PERCENTAGE.
      MESSAGE LV_TEXT TYPE 'I'.

      WHEN 'PERCENTAGE_MARKED'.
        GO_ALV->GET_SELECTED_ROWS(
        IMPORTING
          ET_ROW_NO = LT_ROW_NO ). "### ## #### ##"

      IF LINES( LT_ROW_NO ) > 0.
        LOOP AT LT_ROW_NO INTO LS_ROW_NO.
        READ TABLE GT_SFLIGHT INTO GS_SFLIGHT
        INDEX LS_ROW_NO-ROW_ID.

        LV_OCCUPIED = LV_OCCUPIED + GS_SFLIGHT-SEATSOCC.
        LV_CAPACITY = LV_CAPACITY + GS_SFLIGHT-SEATSMAX.

        ENDLOOP.

      LV_PERCENTAGE = LV_OCCUPIED / LV_CAPACITY * 100.
      MOVE LV_PERCENTAGE TO LV_TEXT.

      LV_TEXT =
      'PERCENTAGE OF OCCUPIED SEATS : '(OCC) && ' ' && LV_TEXT.

      MESSAGE LV_TEXT TYPE 'I'.
      ELSE.
        MESSAGE 'PLEASE MARK AT LEAST ONE ROW'(ONE) TYPE 'I'.
      ENDIF.

      ENDCASE.





  ENDMETHOD.
ENDCLASS.
